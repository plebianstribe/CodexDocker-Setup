# 1) Stage your local Codex config into this build context (required for COPY)
cp ~/.codex/config.toml ./codex-config.toml

# 2) Build image (pyenv variant)
docker build \
  -f DockerWithPyEnv/Dockerfile \
  --build-arg CUDA_TAG=12.1.1-cudnn8-runtime-ubuntu22.04 \
  --build-arg TORCH_CUDA_CHANNEL=cu121 \
  --build-arg PYTHON_VERSION=3.12.12 \
  --build-arg PYENV_ENV_NAME=mainEnv \
  -t codex-pyenv:latest .

# 3) Run container (bind local persistent pyenv root)
echo "Host home dir: $HOME"
echo "Repo dir mounted to /workspace: $PWD"
docker run --rm -it \
  --name codex-pyenv \
  --gpus all \
  --env-file "$PWD/.env" \
  --volume "$PWD":/workspace \
  --volume "$PWD/docker_pyenv_root":/root/.pyenv \
  codex-pyenv:latest

# 4) One-shot runtime check (pyenv env + torch)
docker run --rm -it \
  --gpus all \
  --env-file "$PWD/.env" \
  --volume "$PWD":/workspace \
  --volume "$PWD/docker_pyenv_root":/root/.pyenv \
  codex-pyenv:latest \
  bash -lc 'python --version && which python && python -c "import torch; print(torch.__version__); print(torch.cuda.is_available())"'

# 5) Run pyenv image with pre-mapped non-standard host ports and both repos mounted
docker run --rm -it \
  --name codex-pyenv \
  --gpus all \
  --env-file "$PWD/.env" \
  --volume "$PWD":/workspace \
  --volume "$PWD/docker_pyenv_root":/root/.pyenv \
  --volume "$HOME/path/to/other-repo":/app \
  -p 18888:8888 \
  -p 18000:8000 \
  -p 13000:3000 \
  -p 15173:5173 \
  -p 17860:7860 \
  -p 18501:8501 \
  -p 18080:8080 \
  -p 11434:11434 \
  codex-pyenv:latest

# 6) Start JupyterLab in pyenv main environment and browse both /workspace and /app
# Open from host: http://localhost:18888
docker run --rm -it \
  --name codex-pyenv-jupyter \
  --gpus all \
  --env-file "$PWD/.env" \
  --volume "$PWD":/workspace \
  --volume "$PWD/docker_pyenv_root":/root/.pyenv \
  --volume "$HOME/path/to/other-repo":/app \
  -p 18888:8888 \
  codex-pyenv:latest \
  bash -lc 'python -m jupyter lab --ip=0.0.0.0 --port=8888 --no-browser --allow-root --ServerApp.root_dir=/ --ServerApp.token="" --ServerApp.password=""'
